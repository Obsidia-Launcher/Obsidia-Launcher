// Assign random flex-basis to gallery images on page load for dynamic sizing document.addEventListener('DOMContentLoaded', function() { const imgs = document.querySelectorAll('.gallery-images img'); imgs.forEach(img => { // Random flex-basis between 28% and 32% const percent = (Math.random() * 4 + 28).toFixed(2); img.style.flexBasis = percent + '%'; img.classList.add('gallery-img-random'); }); });
