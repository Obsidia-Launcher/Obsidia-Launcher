// Assign random sizes to gallery images on page load document.addEventListener('DOMContentLoaded', function() { const imgs = document.querySelectorAll('.gallery-images img'); imgs.forEach(img => { // Random width between 180px and 320px const w = Math.floor(Math.random() * (320 - 180 + 1)) + 180; img.style.width = w + 'px'; img.classList.add('gallery-img-random'); }); });
